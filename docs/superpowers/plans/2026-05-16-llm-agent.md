**Architecture:** `LLMAgent` implements the same `act(infoset)` interface as `DeepAgent`. A new Flask endpoint assembles LLMAgent/DeepAgent per configurable positions and feeds them to the existing `generate_ai_battle_data` game loop.

**Tech Stack:** Python (openai package for API calls), React (thin wrapper over GamePlaybackView)

"""Configuration for LLMAgent, read from environment variables."""

import os


def get_llm_config():
    api_key = os.environ.get("DEEPSEEK_API_KEY", "")
    if not api_key:
        raise ValueError("DEEPSEEK_API_KEY environment variable is required")

    raw_positions = os.environ.get("LLM_AGENT_POSITIONS", "0,1,2")
    llm_positions = [int(p.strip()) for p in raw_positions.split(",")]

    return {
        "api_key": api_key,
        "base_url": os.environ.get("DEEPSEEK_BASE_URL", "https://api.deepseek.com"),
        "model": os.environ.get("LLM_MODEL", "deepseek-v4-pro"),
        "timeout": int(os.environ.get("LLM_TIMEOUT", "30")),
        "max_retries": int(os.environ.get("LLM_MAX_RETRIES", "3")),
        "llm_agent_positions": llm_positions,
    }

```python
"""LLMAgent: Dou Dizhu player powered by LLM via HTTP API."""

import json
import logging
import time
from typing import List, Tuple

from openai import OpenAI

from card_maps import EnvCard2RealCard, RealCard2EnvCard
from game import InfoSet
from llm_config import get_llm_config

logger = logging.getLogger(__name__)

SYSTEM_PROMPT = """\
You are an expert Dou Dizhu (Chinese Poker) player. Analyze the game state and choose the best move from the provided legal actions.

## Card Encoding
Each card is a single character: 3, 4, 5, 6, 7, 8, 9, T(10), J, Q, K, A, 2, X(Small Joker), D(Big Joker).

## Game Rules Summary
- 54-card deck. 3 players: one landlord (20 cards) vs two peasants (17 cards each).
- Card ranks (low to high): 3 < 4 < 5 < 6 < 7 < 8 < 9 < T < J < Q < K < A < 2 < X < D
- Move types: single, pair, triple, triple+single, triple+pair, bomb (4 of a kind), king bomb (X+D), serial single (5+ consecutive), serial pair (3+ consecutive pairs), serial triple (2+ consecutive triples), serial triple+single, serial triple+pair, four+two singles, four+two pairs.
- You must beat the last opponent move with a higher move of the same type, OR play a bomb/king bomb to override. Bombs beat non-bomb moves. King bomb beats all.
- You may pass even if you have a legal move.
- Goal: be the first to empty your hand.

## Output Format
Respond with a JSON object only:
{"analysis": "brief strategy reasoning", "action": "<card_string>", "confidence": 0.0-1.0}

- "action": concatenated card codes for the chosen move, e.g. "345" for a 3-4-5 straight, "33" for a pair of 3s.
- "action": "" (empty string) for pass.
- Pick ONLY from the provided legal actions list."""


class LLMAgent:
    """Dou Dizhu agent that uses an LLM API for move decisions."""

    ROLE_NAMES = {0: "landlord", 1: "peasant", 2: "peasant"}

    def __init__(self, position: int):
        if position not in (0, 1, 2):
            raise ValueError(f"position must be 0, 1, or 2, got {position}")
        self.position = position
        config = get_llm_config()
        self._client = OpenAI(api_key=config["api_key"], base_url=config["base_url"])
        self._model = config["model"]
        self._timeout = config["timeout"]
        self._max_retries = config["max_retries"]

    # ------------------------------------------------------------------
    # Card format conversion
    # ------------------------------------------------------------------

    @staticmethod
    def _env_action_to_str(action: List[int]) -> str:
        """Convert env-format action to display string. [] -> '' (pass)."""
        if not action:
            return ""
        return "".join(EnvCard2RealCard[c] for c in action)

    @staticmethod
    def _str_to_env_action(action_str: str) -> List[int]:
        """Convert display string back to env-format action. '' or 'pass' -> []."""
        if not action_str or action_str.lower() == "pass":
            return []
        return [RealCard2EnvCard[c] for c in action_str]

    # ------------------------------------------------------------------
    # Prompt building
    # ------------------------------------------------------------------

    def _build_user_message(self, infoset: InfoSet) -> str:
        role = self.ROLE_NAMES[self.position]

        hand_str = " ".join(EnvCard2RealCard[c] for c in sorted(infoset.player_hand_cards, reverse=True))

        cards_left = ", ".join(f"Player {i}: {infoset.num_cards_left[i]}" for i in range(3))

        rival_str = self._env_action_to_str(infoset.rival_move) if infoset.rival_move else "pass"

        history_lines = []
        for i, action in enumerate(infoset.card_play_action_seq):
            player_idx = i % 3
            action_str = self._env_action_to_str(action) if action else "pass"
            history_lines.append(f"  Player {player_idx}: {action_str}")
        history_str = "\n".join(history_lines) if history_lines else "  (none)"

        legal_str = ", ".join(
            self._env_action_to_str(a) if a else "pass" for a in infoset.legal_actions
        )

        return f"""## Your Position
Player {self.position} ({role})

## Your Hand
{hand_str}

## Cards Remaining
{cards_left}

## Bombs Played
{infoset.bomb_num}

## Last Opponent Move
{rival_str}

## Play History
{history_str}

## Legal Actions
{legal_str}

Choose the single best action from the legal actions list above."""

    # ------------------------------------------------------------------
    # LLM API call
    # ------------------------------------------------------------------

    def _call_llm(self, messages: list) -> str:
        """Call LLM API with retries. Returns content string."""
        last_error = None
        for attempt in range(self._max_retries):
            try:
                response = self._client.chat.completions.create(
                    model=self._model,
                    messages=messages,
                    response_format={"type": "json_object"},
                    reasoning_effort="high",
                    extra_body={"thinking": {"type": "enabled"}},
                    timeout=self._timeout,
                )
                return response.choices[0].message.content
            except Exception as e:
                last_error = e
                if attempt < self._max_retries - 1:
                    wait = 2**attempt
                    logger.warning(
                        "LLMAgent position=%d: API call attempt %d failed: %s. Retrying in %ds...",
                        self.position, attempt + 1, e, wait,
                    )
                    time.sleep(wait)

        raise last_error

    # ------------------------------------------------------------------
    # Response parsing
    # ------------------------------------------------------------------

    def _parse_response(self, content: str, legal_actions: List[List[int]]) -> Tuple[List[int], float]:
        """Parse LLM JSON response, validate action, return (env_action, confidence)."""
        try:
            data = json.loads(content)
        except json.JSONDecodeError as e:
            raise ValueError(f"Failed to parse LLM response as JSON: {e}") from e

        action_str = data.get("action", "")
        confidence = float(data.get("confidence", 0.5))

        env_action = self._str_to_env_action(action_str)

        if env_action not in legal_actions:
            raise ValueError(f"LLM returned illegal action '{action_str}' ({env_action}) not in legal actions")

        return env_action, confidence

    # ------------------------------------------------------------------
    # Fallback
    # ------------------------------------------------------------------

    def _fallback_action(self, legal_actions: List[List[int]]) -> Tuple[List[List[int]], List[float]]:
        logger.warning(
            "LLMAgent position=%d: LLM call failed, falling back to first legal action. "
            "Legal actions count=%d, first=%s",
            self.position, len(legal_actions),
            self._env_action_to_str(legal_actions[0]) if legal_actions else "none",
        )
        action = legal_actions[0] if legal_actions else []
        return [action], [0.0]

    # ------------------------------------------------------------------
    # Main interface
    # ------------------------------------------------------------------

    def act(self, infoset: InfoSet) -> Tuple[List[List[int]], List[float]]:
        """Choose the best action for the given game state."""
        messages = [
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": self._build_user_message(infoset)},
        ]

        try:
            content = self._call_llm(messages)
            env_action, confidence = self._parse_response(content, infoset.legal_actions)
            return [env_action], [confidence]
        except Exception:
            logger.exception("LLMAgent position=%d: act() failed", self.position)
            return self._fallback_action(infoset.legal_actions)
```

    """Get agent list for LLM battle, configurable via LLM_AGENT_POSITIONS env var."""
    pretrained_dir = os.path.join(os.path.dirname(os.path.abspath(__file__)), "pretrained", "douzero_pretrained")

    llm_positions = config["llm_agent_positions"]

import React, { useCallback } from 'react';
import { useTranslation } from 'react-i18next';

        <GamePlaybackView
            fetchData={fetchData}
            validate={null}
            autoStart={false}
            gameBoardProps={{ gamePlayable: false, showCardBack: false }}
            startLabel={t('game_playback.new_llm_battle')}
            errorMessage={t('errors.error_generating_battle')}
        />
    );
}

export default LLMBattleView;
